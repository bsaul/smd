Using the smd() function

Bradley Saul

2018-11-01

The SugarMaryDenver package provides the smd method to compute standardized mean differences between two groups for continuous values (numeric and integer data types) and categorical values (factor, character, and logical). The method also works on matrix, list, and data.frame data types by applying smd() over the columns of the matrix or data.frame and each item of the list. The package is based on Yang and Dalton (2012).

The smd function computes the standardized mean difference for each level \(k\) of a grouping variable compared to a reference \(r\) level:

\[ d_k = \sqrt{(\bar{x}_r - \bar{x}_{k})^{\intercal}S_{rk}^{-1}(\bar{x}_r - \bar{x}_{k})} \]

where \(\bar{x}_{\cdot}\) and \(S_{rk}\) are the sample mean and covariances for reference group \(r\) and group \(k\), respectively. In the case that \(x\) is categorical, \(\bar{x}\) is the vector of proportions of each category level within a group, and \(S_{rk}\) is the multinomical covariance matrix.

Standard errors are computed using the formula described in Hedges and Olkin (1985):

\[ \sqrt{ \frac{n_r + n_k}{n_rn_k} + \frac{d_k^2}{2(n_r + n_k)} } \]

Examples

library(SugarMaryDenver)
set.seed(123)
xn <- rnorm(90)
gg2 <- rep(LETTERS[1:2], each = 45)
gg3 <- rep(LETTERS[1:3], each = 30)

smd(x = xn, g = gg2)
#>   term   estimate
#> 1    B 0.03375131
smd(x = xn, g = gg3)
#>   term    estimate
#> 1    B -0.24746529
#> 2    C -0.07714974
smd(x = xn, g = gg2, std.error = TRUE)
#>   term   estimate std.error
#> 1    B 0.03375131 0.2108335
smd(x = xn, g = gg3, std.error = TRUE)
#>   term    estimate std.error
#> 1    B -0.24746529 0.2591852
#> 2    C -0.07714974 0.2582949
xi <- sample(1:20, 90, replace = TRUE)
smd(x = xi, g = gg2)
#>   term     estimate
#> 1    B -0.008289296
xc <- unlist(replicate(2, sort(sample(letters[1:3], 45, replace = TRUE)), simplify = FALSE))
smd(x = xc, g = gg2)
#>   term  estimate
#> 1    B 0.2518824
xf <- factor(xc)
smd(x = xf, g = gg2)
#>   term  estimate
#> 1    B 0.2518824
xl <- as.logical(rbinom(90, 1, prob = 0.5))
smd(x = xl, g = gg2)
#>   term    estimate
#> 1    B -0.04435493
mm <- cbind(xl, xl, xl, xl)
smd(x = mm, g = gg3, std.error = FALSE)
#>               xl          xl          xl          xl
#> [1,]  0.06744642  0.06744642  0.06744642  0.06744642
#> [2,] -0.26514719 -0.26514719 -0.26514719 -0.26514719
ll <- list(xn = xn, xi = xi, xf = xf, xl = xl)
smd(x = ll, g = gg3)
#>   variable term    estimate
#> 1       xn    B -0.24746529
#> 2       xn    C -0.07714974
#> 3       xi    B  0.12351572
#> 4       xi    C -0.05652726
#> 5       xf    B  1.63815271
#> 6       xf    C  2.39588392
#> 7       xl    B  0.06744642
#> 8       xl    C -0.26514719
df <- data.frame(xn, xi, xc, xf, xl)
smd(x = df, g = gg3)
#>    variable term    estimate
#> 1        xn    B -0.24746529
#> 2        xn    C -0.07714974
#> 3        xi    B  0.12351572
#> 4        xi    C -0.05652726
#> 5        xc    B  1.63815271
#> 6        xc    C  2.39588392
#> 7        xf    B  1.63815271
#> 8        xf    C  2.39588392
#> 9        xl    B  0.06744642
#> 10       xl    C -0.26514719

Using smd with dplyr

library(dplyr)
df$g <- gg2
df %>%
  summarize_at(
    .vars = vars(dplyr::matches("^x")),
    .funs = funs(smd = smd(., g = g)$estimate))
#>       xn_smd       xi_smd    xc_smd    xf_smd      xl_smd
#> 1 0.03375131 -0.008289296 0.2518824 0.2518824 -0.04435493
